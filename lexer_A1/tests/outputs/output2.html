using namespace std;

// Structure of a node 
struct Node {
    int data;
    Node* next;
};

// Function to get a new node 
Node* getNode(int data)
{
    // allocate space 
    Node* newNode = new Node;

    // put in data 
    newNode->data = data;
    newNode->next = NULL;
    return newNode;
}

// of linked list in reverse order
